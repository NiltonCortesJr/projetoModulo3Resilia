SELECT
r.driverId,
concat(forename,' ',surname) as Piloto,
nationality,
position,
count(*) as numero_podio

FROM
mod3_formula1.drivers as d

INNER JOIN
mod3_formula1.results as r

ON
d.driverId = r.driverId

WHERE
position in (1, 2, 3)

GROUP BY
driverId

ORDER BY
numero_podio DESC

LIMIT
10
