-- Querie lista os 5 pilotos que mais obtiveram as voltas mais r√°pidas.

SELECT
	laptimes.driverId,
	concat(forename, " ", surname),
	nationality,
	time,
	milliseconds

FROM 
	mod3_formula1.laptimes

INNER JOIN
	drivers

ON
	drivers.driverId = laptimes.driverId

ORDER BY
	time, milliseconds

LIMIT
	5;
