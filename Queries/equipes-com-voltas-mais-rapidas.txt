-- Querie lista os 5 equipes que mais obtiveram voltas mais r√°pidas.

SELECT
	name,
	nationality,
	fastestLapTime

FROM 
 
 	mod3_formula1.results as r
 
INNER JOIN
 
 	mod3_formula1.constructors as c
 
ON
 
 	r.constructorId = c.constructorId
 
HAVING
 	fastestLapTime is not null
 
ORDER BY
 	fastestLapTime
LIMIT
	5;
